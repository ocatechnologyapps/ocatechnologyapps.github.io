  <script>
    // --- Simple SPA routing by hash ---
    const views = document.querySelectorAll('.view');
    function showRoute(route){
      views.forEach(v=>v.style.display='none');
      const el = document.getElementById(route);
      if(el) el.style.display='block';
      document.querySelectorAll('a.nav').forEach(a=>a.classList.toggle('active', a.dataset.route===route));
    }
    function handleHash(){
      const h = location.hash.replace('#','') || 'home';
      // special: if hash matches app detail like app-1
      if(h.startsWith('app-')) return showAppDetail(h);
      showRoute(h);
      // update nav active
    }
    window.addEventListener('hashchange', handleHash);
    handleHash();

    // set year
    document.getElementById('year').textContent = new Date().getFullYear();

    // --- Simple app storage (localStorage) ---
    const STORAGE_KEY = 'ocat_apps_v1';
    function defaultApps(){
      return [{id:'app-1',title:'Empanadas Galácticas',type:'Juego - Clicker',desc:'Vende empanadas en Marte. Cada click = 10 coins.',playId:'',policy:'Política por defecto: no recopilamos datos personales sin consentimiento.'}];
    }
    function loadApps(){
      try{
        const raw = localStorage.getItem(STORAGE_KEY);
        return raw? JSON.parse(raw): defaultApps();
      }catch(e){return defaultApps();}
    }
    function saveApps(list){localStorage.setItem(STORAGE_KEY, JSON.stringify(list));renderApps()}

    function renderApps(){
      const list = loadApps();
      const container = document.getElementById('appsList');
      container.innerHTML = '';
      list.forEach(app=>{
        const div = document.createElement('div'); div.className='app-card'; div.dataset.id=app.id;
        div.innerHTML = 
          <div class="app-thumb">${app.title.split(' ').map(s=>s[0]).slice(0,2).join('')}</div>
          <div class="app-info">
            <div style="display:flex;justify-content:space-between;align-items:center">
              <div>
                <strong>${app.title}</strong>
                <div class="muted small">${app.type}</div>
              </div>
              <div class="app-actions">
                <a class="btn" href="#${app.id}" style="padding:8px 10px">Ver</a>
                <button class="btn" onclick="copyPolicyLink('${app.id}')">Política</button>
              </div>
            </div>
            <div class="muted small" style="margin-top:8px">${new Date().toISOString().slice(0,10)}</div>
          </div>
        ;
        container.appendChild(div);
      })
    }
    renderApps();

    // --- App detail renderer ---
    function showAppDetail(id){
      showRoute('apps');
      const app = loadApps().find(a=>a.id===id);
      const out = document.getElementById('appDetail');
      if(!app){out.innerHTML='<div class="card">App no encontrada</div>';return}
      out.innerHTML = 
        <div class="card">
          <div style="display:flex;justify-content:space-between;align-items:center">
            <div>
              <h3 style="margin:0">${app.title}</h3>
              <div class="muted small">${app.type} • ID: ${app.playId||'no definido'}</div>
            </div>
            <div>
              <button class="btn" onclick="location.hash='#apps'">Volver</button>
            </div>
          </div>
          <div style="margin-top:10px">${app.desc}</div>
          <hr style="margin:12px 0">
          <h4>Política de privacidad</h4>
          <div class="policy">${app.policy.replace(/\n/g,'<br>')}</div>
        </div>
      ;
    }
function copyPolicyLink(id){
      const url = location.origin + location.pathname + '#'+id;
      navigator.clipboard?.writeText(url).then(()=>alert('Enlace de política copiado al portapapeles'));


    }

    // --- Admin ---
    const ADMIN_PASS_KEY = 'ocat_admin_pass';
    function loginAdmin(){
      const pass = document.getElementById('adminPass').value || '';
      const real = localStorage.getItem(ADMIN_PASS_KEY) || 'ocatoken';
      if(pass===real){
        document.getElementById('adminAuth').style.display='none';
        document.getElementById('adminPanel').style.display='block';
      } else { alert('Contraseña incorrecta') }
    }
    function logoutAdmin(){
      document.getElementById('adminPanel').style.display='none';
      document.getElementById('adminAuth').style.display='block';
    }
    function addApp(){
      const title = document.getElementById('newTitle').value.trim();
      if(!title){alert('El título es obligatorio');return}
      const type = document.getElementById('newType').value.trim()||'App';
      const desc = document.getElementById('newDesc').value.trim()||'';
      const playId = document.getElementById('newPlayId').value.trim()||'';
      const policy = document.getElementById('newPolicy').value.trim()||'Política por defecto.';
      const list = loadApps();
      const id = 'app-'+(list.length+1);
      list.push({id,title,type,desc,playId,policy});
      saveApps(list);
      alert('App agregada');
      document.getElementById('newTitle').value='';document.getElementById('newType').value='';document.getElementById('newDesc').value='';document.getElementById('newPolicy').value='';
    }

    // --- initialize route on load ---
    if(location.hash) handleHash();
  </script>
